% drawing the actual BG
\newcommand{\drawBackground}[1]{
  \AddToShipoutPictureBG{%
    \begin{tikzpicture}[remember picture,overlay]
      \fill[#1]
      (current page.south west)
      rectangle ([xshift=0.40\paperwidth]current page.north west);
    \end{tikzpicture}%
  }
}

% we want a way to draw a proper title, so we redefine
% requires author to be set
\makeatletter
\renewcommand{\maketitle}{
  \vspace*{0.5cm}
  {\titlefont\fontsize{40pt}{60pt}\selectfont\@author} \\
  \medskip \\
}

      \clip (0,0) circle (2cm);
      \node at (#1,#2) {\includegraphics[width=#3]{your-image.jpg}};
      \draw[line width=1.5pt,color=white] (0,0) circle (2cm);
    \end{tikzpicture}
  \end{center}
}
